                                                                        Bibek Luitel • 4 months ago

                                                                        implementation of using iterative .. O(n ) time and without extra space used

                                                                        void connect( struct node * root){
                                                                        struct node * head= NULL, * prev= NULL;
                                                                        root->nextRight= NULL;
                                                                        if (root==NULL){ return; }
                                                                        while(root){

                                                                        if ( head== NULL && root->left){ head= root->left;}
                                                                        if( head== NULL && root->right){ head= root->right;}
                                                                        if(root->left && root->right){

                                                                        root->left->nextRight= root->right;

                                                                        if(prev){ prev->nextRight=root->left;}

                                                                        prev= root->right;
                                                                        }
                                                                        else if (root->left){

                                                                        if(prev){ prev->nextRight=root->left;}

                                                                        prev= root->left;
                                                                        }
                                                                        else if (root->right){

                                                                        if(prev){ prev->nextRight=root->right;}

                                                                        prev= root->left;

                                                                        }

                                                                        root= root->nextRight;

                                                                        if(root==NULL && head){

                                                                        root= head;
                                                                        head= NULL;

                                                                        if(prev){prev->nextRight= NULL;}

                                                                        }

                                                                        }
                                                                        }

                                                                                A simple implementation making use of earlier methods :p
                                                                                #include<bits stdc++.h="">

                                                                                using namespace std;

                                                                                struct tree{

                                                                                int data;

                                                                                struct tree *left;

                                                                                struct tree *right;

                                                                                struct tree *rightptr;

                                                                                };

                                                                                struct tree* new_node(int x)

                                                                                {

                                                                                struct tree *temp;

                                                                                temp=new struct tree;

                                                                                temp->data=x;

                                                                                temp->left=NULL;

                                                                                temp->right=NULL;

                                                                                temp->rightptr=NULL;

                                                                                }

                                                                                struct tree* make_tree(int a[],int i,int n)

                                                                                {

                                                                                if(i>n)

                                                                                return NULL;

                                                                                struct tree *temp;

                                                                                temp=new_node(a[i]);

                                                                                temp->left=make_tree(a,2*i+1,n);

                                                                                temp->right=make_tree(a,2*i+2,n);

                                                                                return temp;

                                                                                }

                                                                                void print(struct tree *root)

                                                                                {

                                                                                if(root!=NULL)

                                                                                {

                                                                                print(root->left);

                                                                                cout<<root->data;

                                                                                print(root->right);

                                                                                }

                                                                                }

                                                                                int height(struct tree *root)

                                                                                {

                                                                                if(root==NULL)

                                                                                return 0;

                                                                                return 1+max(height(root->left),height(root->right));

                                                                                }

                                                                                struct tree *prev;

                                                                                void pointer(struct tree *root,int lev)

                                                                                {

                                                                                if(root==NULL||lev<1)

                                                                                return ;

                                                                                else if(lev==1)

                                                                                {

                                                                                if(prev==NULL)

                                                                                prev=root;

                                                                                else

                                                                                {

                                                                                prev->rightptr=root;

                                                                                prev=root;

                                                                                }

                                                                                }

                                                                                else

                                                                                {

                                                                                pointer(root->left,lev-1);

                                                                                pointer(root->right,lev-1);

                                                                                }

                                                                                }

                                                                                void connect(struct tree *root)

                                                                                {

                                                                                if(root==NULL)

                                                                                return ;

                                                                                int h=height(root);

                                                                                for(int i=1;i<=h;i++)

                                                                                {

                                                                                prev=NULL;

                                                                                pointer(root,i);

                                                                                }

                                                                                }

                                                                                int main()

                                                                                {

                                                                                int a[]={1,2,3,4,5,6,7};

                                                                                struct tree *root;

                                                                                root=make_tree(a,0,6);

                                                                                connect(root);

                                                                                return 0;

                                                                                }

                                                                                                                                                      Optimized recursive code:
                                                                                                                                                      1) Removed unnecessary else part recursion (as the first recursion always takes care of that in previous steps)
                                                                                                                                                      2) If the nodes are already processed then dont process them.

                                                                                                                                                      void join_level(node *root)
                                                                                                                                                      {
                                                                                                                                                      if (!root)
                                                                                                                                                      return;

                                                                                                                                                      if (root->left && root->left->nextdata)
                                                                                                                                                      return;

                                                                                                                                                      if (root->right && root->right->nextdata)
                                                                                                                                                      return;

                                                                                                                                                      if (root->nextdata)
                                                                                                                                                      join_level(root->nextdata);

                                                                                                                                                      if (root->left)
                                                                                                                                                      {
                                                                                                                                                      if (root->right)
                                                                                                                                                      {
                                                                                                                                                      root->left->nextdata = root->right;
                                                                                                                                                      root->right->nextdata = getnext(root);
                                                                                                                                                      }
                                                                                                                                                      else
                                                                                                                                                      root->left->nextdata = getnext(root);

                                                                                                                                                      join_level(root->left);
                                                                                                                                                      }
                                                                                                                                                      else if (root->right)
                                                                                                                                                      {
                                                                                                                                                      root->right->nextdata = getnext(root);
                                                                                                                                                      join_level(root->right);
                                                                                                                                                      }
                                                                                                                                                      }
