                                            struct node* new;

                                            new->data=x;
                                            new->axis=0;
                                            new->left=NULL;
                                            new->right=NULL;
                                            return new;
                                            }

                                            void indexo(struct node* root,int level)
                                            {
                                            if(root==NULL)return;
                                            root->axis=level;
                                            indexo(root->left,level-1);
                                            indexo(root->right,level+1);
                                            }

                                            int max(int a,int b)
                                            {
                                            if(a>=b)return a;
                                            return b;
                                            }

                                            int min(int a, int b)
                                            {
                                            if(a<=b)return a;
                                            return b;
                                            }

                                            int lefto(struct node* root)
                                            {
                                            if(root==NULL)return 0;
                                            return min(root->axis,min(lefto(root->left),lefto(root->right)));
                                            }

                                            int righto(struct node* root)
                                            {
                                            if(root==NULL)return 0;
                                            return max(root->axis,max(righto(root->left),righto(root->right)));
                                            }

                                            void doSum(int s[],int n[],struct node* root,int l)
                                            {
                                            if(root==NULL)return;
                                            int t;
                                            t=root->axis - l + 1;
                                            n[t]++;
                                            s[t]+=root->data;
                                            doSum(s,n,root->left,l);
                                            doSum(s,n,root->right,l);
                                            }

                                                                              struct tree{

                                                                              int data;

                                                                              int val;

                                                                              struct tree *left;

                                                                              struct tree *right;

                                                                              };

                                                                              struct tree* new_node(int x)

                                                                              {

                                                                              struct tree *temp;

                                                                              temp=new struct tree;

                                                                              temp->data=x;

                                                                              temp->left=NULL;

                                                                              temp->right=NULL;

                                                                              }

                                                                              struct tree* make_tree(int a[],int i,int n)

                                                                              {

                                                                              if(i>n)

                                                                              return NULL;

                                                                              struct tree *temp;

                                                                              temp=new_node(a[i]);

                                                                              temp->left=make_tree(a,2*i+1,n);

                                                                              temp->right=make_tree(a,2*i+2,n);

                                                                              return temp;

                                                                              }

                                                                              void print(struct tree *root)

                                                                              {

                                                                              if(root!=NULL)

                                                                              {

                                                                              print(root->left);

                                                                              cout<<root->data;

                                                                              print(root->right);

                                                                              }

                                                                              }

                                                                              map<int,int> sum;

                                                                              void assign(struct tree *root,int k)

                                                                              {

                                                                              if(root==NULL)

                                                                              return ;

                                                                              root->val=k;

                                                                              assign(root->left,k-1);

                                                                              assign(root->right,k-1);

                                                                              }

                                                                              void cal(struct tree *root)

                                                                              {

                                                                              if(root!=NULL)

                                                                              {

                                                                              cal(root->left);

                                                                              map<int,int> ::iterator it;

                                                                              it=sum.find(root->val);

                                                                              if(it!=sum.end())

                                                                              it->second+=root->data;

                                                                              else

                                                                              sum.insert(std::pair<int,int>(root->val,root->data));

                                                                              cal(root->right);

                                                                              }

                                                                              }

                                                                              void printsum()

                                                                              {

                                                                              std::map<int,int> ::iterator it;

                                                                              for(it=sum.begin(),it!=sum.end();it++)

                                                                              cout<<it->second<<" ";

                                                                              }

                                                                              {

                                                                              int a[]={1,2,3,4,5,6,7};

                                                                              struct tree *root;

                                                                              root=make_tree(a,0,6);

                                                                              assign(root,0);

                                                                              cal(root);

                                                                              return 0;

                                                                                                                                                                                    #include <stdio.h>

                                                                                                                                                                                    #include <stdlib.h>

                                                                                                                                                                                    struct node{

                                                                                                                                                                                    int data;

                                                                                                                                                                                    struct node* left;

                                                                                                                                                                                    struct node* right;

                                                                                                                                                                                    int val;

                                                                                                                                                                                    };

                                                                                                                                                                                    struct node* createNode(int k)

                                                                                                                                                                                    {

                                                                                                                                                                                    struct node* node=(struct node*)malloc(sizeof(struct node));

                                                                                                                                                                                    node->data=k;

                                                                                                                                                                                    node->left=NULL;

                                                                                                                                                                                    node->right=NULL;

                                                                                                                                                                                    return node;

                                                                                                                                                                                    }

                                                                                                                                                                                    void horiDisRec(struct node* root,int val)

                                                                                                                                                                                    {

                                                                                                                                                                                    if (root==NULL) {

                                                                                                                                                                                    return;

                                                                                                                                                                                    }

                                                                                                                                                                                    else

                                                                                                                                                                                    root->val=val;

                                                                                                                                                                                    horiDisRec(root->left, val-1);

                                                                                                                                                                                    horiDisRec(root->right, val+1);

                                                                                                                                                                                    return;

                                                                                                                                                                                    }

                                                                                                                                                                                    void horiDistance(struct node* root)

                                                                                                                                                                                    {

                                                                                                                                                                                    /*calls the recursive function to set values of horizontal distance for each node*/

                                                                                                                                                                                    horiDisRec(root,0);

                                                                                                                                                                                    return;

                                                                                                                                                                                    }

                                                                                                                                                                                    void traversal(struct node* root,int a[],int i,int n)

                                                                                                                                                                                    {

                                                                                                                                                                                    if (root==NULL) {

                                                                                                                                                                                    return;

                                                                                                                                                                                    }

                                                                                                                                                                                    /*preorder traversal to sum data of nodes for each horizontal distance*/

                                                                                                                                                                                    a[root->val-i]+=root->data;

                                                                                                                                                                                    traversal(root->left, a, i, n);

                                                                                                                                                                                    traversal(root->right, a,i, n);

                                                                                                                                                                                    return;

                                                                                                                                                                                    }

                                                                                                                                                                                    void printVertical(struct node* root)

                                                                                                                                                                                    {

                                                                                                                                                                                    if (root==NULL) {

                                                                                                                                                                                    return;

                                                                                                                                                                                    }

                                                                                                                                                                                    int j=0;

                                                                                                                                                                                    int i=0,n=0;

                                                                                                                                                                                    struct node* l=root;

                                                                                                                                                                                    struct node* r=root;

                                                                                                                                                                                    while (l->left!=NULL) {

                                                                                                                                                                                    l=l->left;

                                                                                                                                                                                    }

                                                                                                                                                                                    i=l->val;//Horizontal Distance of the left mode node

                                                                                                                                                                                    while (r->right!=NULL) {

                                                                                                                                                                                    r=r->right;

                                                                                                                                                                                    }

                                                                                                                                                                                    n=r->val;//Horizontal Distance of the right most node

                                                                                                                                                                                    int arr[n-i+1];//array to store the sum for each horizontal distance

                                                                                                                                                                                    /*loop initialising each element of array initially to zero*/

                                                                                                                                                                                    for (j=0; j<=(n-i); j++) {

                                                                                                                                                                                    arr[j]=0;

                                                                                                                                                                                    }

                                                                                                                                                                                    /* function to sum values for each horizontal distance available*/

                                                                                                                                                                                    traversal(root,arr,i,n);

                                                                                                                                                                                    /*prints the array*/

                                                                                                                                                                                    for (j=0; j<=(n-i); j++) {

                                                                                                                                                                                    printf("%d ",arr[j]);

                                                                                                                                                                                    }

                                                                                                                                                                                    return;

                                                                                                                                                                                    }

                                                                                                                                                                                    int main()

                                                                                                                                                                                    {
