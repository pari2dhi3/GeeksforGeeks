                                                                                                if (root == null)
                                                                                                return 0;

                                                                                                return level + 1;

                                                                                                int finalLevel = 0;

                                                                                                finalLevel = findLevel(root.right, target, ++level);
                                                                                                else
                                                                                                finalLevel = findLevel(root.left, target, ++level);

                                                                                                }
