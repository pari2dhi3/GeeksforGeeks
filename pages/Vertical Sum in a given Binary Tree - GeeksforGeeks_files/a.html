                                            {
                                            struct node* new;
                                            new=(struct node*)malloc(sizeof(struct node));
                                            new->data=x;
                                            new->axis=0;
                                            new->left=NULL;
                                            new->right=NULL;
                                            return new;
                                            }

                                            void indexo(struct node* root,int level)
                                            {
                                            if(root==NULL)return;
                                            root->axis=level;
                                            indexo(root->left,level-1);
                                            indexo(root->right,level+1);
                                            }

                                            int max(int a,int b)
                                            {
                                            if(a>=b)return a;
                                            return b;
                                            }

                                            int min(int a, int b)
                                            {
                                            if(a<=b)return a;
                                            return b;
                                            }

                                            int lefto(struct node* root)
                                            {
                                            if(root==NULL)return 0;
                                            return min(root->axis,min(lefto(root->left),lefto(root->right)));
                                            }

                                            int righto(struct node* root)
                                            {
                                            if(root==NULL)return 0;
                                            return max(root->axis,max(righto(root->left),righto(root->right)));
                                            }

                                            void doSum(int s[],int n[],struct node* root,int l)
                                            {
                                            if(root==NULL)return;
                                            int t;
                                            t=root->axis - l + 1;
                                            n[t]++;
                                            s[t]+=root->data;
                                            doSum(s,n,root->left,l);
                                            doSum(s,n,root->right,l);
                                            }

                                            void vertical_sum(struct node* root)
                                            {
                                            indexo(root,0);
                                            int l,r,i,d;
                                            l=lefto(root);
                                            r=righto(root);
                                            d=r-l+1;
                                            int sum[100]={0};
                                            int num[100]={0};
                                            doSum(sum,num,root,l);
                                            printf("The tree has %d vertical lines\n",d);
                                            for(i=1;i<=d;i++)
                                            printf("Vertical-Line-%d has %d nodes => vertical sum is %d \n",i,num[i],sum[i]);
                                            }

                                                                              struct tree{

                                                                              int data;

                                                                              int val;

                                                                              struct tree *temp;

                                                                              temp=new struct tree;

                                                                              temp->data=x;

                                                                              temp->left=NULL;

                                                                              }

                                                                              struct tree* make_tree(int a[],int i,int n)

                                                                              {

                                                                              if(i>n)

                                                                              return NULL;

                                                                              struct tree *temp;

                                                                              temp=new_node(a[i]);

                                                                              temp->left=make_tree(a,2*i+1,n);

                                                                              temp->right=make_tree(a,2*i+2,n);

                                                                              return temp;

                                                                              }

                                                                              {

                                                                              if(root!=NULL)

                                                                              void assign(struct tree *root,int k)

                                                                              if(root==NULL)

                                                                              return ;

                                                                              root->val=k;

                                                                              assign(root->left,k-1);

                                                                              assign(root->right,k-1);

                                                                              }

                                                                              void cal(struct tree *root)

                                                                              {

                                                                              if(root!=NULL)

                                                                              {

                                                                              cal(root->left);

                                                                              it->second+=root->data;

                                                                              }

                                                                              }

                                                                                                                                                                                    horiDisRec(root,0);

                                                                                                                                                                                    return;

                                                                                                                                                                                    }

                                                                                                                                                                                    /*preorder traversal to sum data of nodes for each horizontal distance*/

                                                                                                                                                                                    a[root->val-i]+=root->data;

                                                                                                                                                                                    traversal(root->left, a, i, n);

                                                                                                                                                                                    traversal(root->right, a,i, n);

                                                                                                                                                                                    return;

                                                                                                                                                                                    }

                                                                                                                                                                                    void printVertical(struct node* root)

                                                                                                                                                                                    {

                                                                                                                                                                                    if (root==NULL) {

                                                                                                                                                                                    return;

                                                                                                                                                                                    }

                                                                                                                                                                                    int j=0;

                                                                                                                                                                                    int i=0,n=0;

                                                                                                                                                                                    struct node* l=root;

                                                                                                                                                                                    struct node* r=root;

                                                                                                                                                                                    while (l->left!=NULL) {

                                                                                                                                                                                    l=l->left;

                                                                                                                                                                                    }

                                                                                                                                                                                    while (r->right!=NULL) {

                                                                                                                                                                                    r=r->right;
