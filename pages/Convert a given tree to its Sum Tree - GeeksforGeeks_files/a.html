                                                  void convertToSumTree(struct Tree *tree)

                                                  {

                                                  if(tree == NULL) return;

                                                  if(tree->left == NULL && tree->right == NULL) {

                                                  tree->data =0;

                                                  return;

                                                  }

                                                  int a = tree->left ? tree->left->data : 0;

                                                  int b = tree->right ? tree->right->data :0;

                                                  tree->data = a + b;

                                                  convertToSumTree(tree->left);

                                                  convertToSumTree(tree->right);

                                                  a = tree->left ? tree->left->data : 0;

                                                  b = tree->right ? tree->right->data :0;

                                                  tree->data += a + b;

                                                  return;

                                                  }

                                                                                                                                    if(root->left == NULL && root->right == NULL)

                                                                                                                                    return temp;

                                                                                                                                    }

                                                                                                                                    int left = toSumTree(root->left);

                                                                                                                                    int right = toSumTree(root->right);

                                                                                                                                    int temp = root->data;

                                                                                                                                    return (root->data+temp);
