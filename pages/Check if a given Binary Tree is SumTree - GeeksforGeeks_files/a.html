                                      A simpler and recursive version in C that works for both positive and negative numbers

                                      bool sum(struct node *a,int *data)
                                      {
                                      if(a==NULL)
                                      return true;
                                      int left=0;
                                      int right=0;
                                      bool l=true;
                                      bool r=true;
                                      if(a->left)
                                      l=sum(a->left,&left);
                                      if(a->right)
                                      r=sum(a->right,&right);
                                      if(a->left==NULL && a->right == NULL)
                                      {
                                      *data = a->data;
                                      return true;

                                      }
                                      if(l==true && r== true)
                                      {
                                      if(a->data == left+right)
                                      {
                                      *data=a->data +left+right;
                                      return true;

                                      }

                                      else
                                      return false;

                                      }
                                      return false;

                                      }

                                          return p->data;

                                          }

                                          else{

                                          l=sumtree(p->left);

                                          r=sumtree(p->right);

                                          if((l==0&&p->left->data!=0)||(r==0&&p->right->data==0)){

                                          }

                                                                                                                                                              if(tree==null||(tree.left==null&&tree.right==null)){
