                                                                                                                                                      int sum(node *root)
                                                                                                                                                      {
                                                                                                                                                      if(!root) return 0;
                                                                                                                                                      return root->data +
                                                                                                                                                      max( sum(root->left) , sum(root->right) );
                                                                                                                                                      }
