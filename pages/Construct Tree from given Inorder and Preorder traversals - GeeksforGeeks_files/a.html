                          It is very easy and solution is similar.
                          Instead of preIndex, you use postIndex starting from last element of Post-Order.
                          And it should be decreasing in each recursive call.
                          Also order of recursion is slightly changed :
                          For Pre-Order it was :
                          tNode->left = buildTree(in, pre, inStrt, inIndex-1);
                          tNode->right = buildTree(in, pre, inIndex+1, inEnd);

                          For Post-Order it will be :
                          tNode->right = buildTree(in, post, inIndex+1, inEnd);

                          tNode->left = buildTree(in, post, inStrt, inIndex-1);

                          I would suggest you try doing the program.

                                                            if ( arr[1][in] == arr[0][i] ) // arr[1] -> stores preorder

                                                            nd * tmp = newNode ( arr[1][in] ) ;
                                                            if( i > s )
                                                            tmp -> left = fun( s , i-1 , in+1 ) ;

                                                                                public class Method2 {

                                                                                public static Node buildTree(int[] inorder, int[] preorder) {

                                                                                return buildTree(inorder, 0, inorder.length - 1, preorder, 0,
                                                                                preorder.length - 1);
                                                                                }

                                                                                static class Node {
                                                                                int data;

                                                                                public Node(int data) {
                                                                                this.data = data;
                                                                                left = null;
                                                                                right = null;
                                                                                }

                                                                                Node right;
                                                                                Node left;
                                                                                }

                                                                                public static Node buildTree(int[] inorder, int startIn, int endIn,
                                                                                int[] preorder, int startPre, int endPre) {

                                                                                if (endIn < startIn || endPre < startPre) {
                                                                                return null;
                                                                                }

                                                                                Node node = new Node(preorder[startPre]);

                                                                                int i = startIn;

                                                                                for (; i < endIn; i++) {
                                                                                if (inorder[i] == preorder[startPre])
                                                                                break;
                                                                                }

                                                                                node.left = buildTree(inorder, startIn, i - 1, preorder, startPre + 1,
                                                                                startPre + 1 + i - startIn);
                                                                                node.right = buildTree(inorder, i + 1, endIn, preorder, startPre + 1
                                                                                + i - startIn, endPre);

                                                                                return node;

                                                                                }

                                                                                // T(N)=O(N)+2T(N/2) ==> time O(nlogn)
                                                                                // space O(N) + extra O(logn)

                                                                                public static void printInOrder(Node node) {

                                                                                if (node == null) {
                                                                                return;

                                                                                } else {

                                                                                printInOrder(node.left);
                                                                                System.out.print(node.data + " ");
                                                                                printInOrder(node.right);
                                                                                }
                                                                                }

                                                                                }

                                                                                                                              Using HashMultiMap for fast search to reduce time to 0(n) in average.
                                                                                                                              Worst case will be when entire tree is of same value.
                                                                                                                              node * inorder_preorder(int *cur, unordered_multimap<int, int=""> *umap, int *pre, int start, int end)

                                                                                                                              {

                                                                                                                              node *temp;

                                                                                                                              int center = 0;

                                                                                                                              if (start > end)

                                                                                                                              return NULL;

                                                                                                                              temp = new node;

                                                                                                                              temp->data = pre[*cur];

                                                                                                                              auto it = umap->equal_range(pre[*cur]);

                                                                                                                              unordered_multimap<int, int="">::iterator k_it = it.first;

                                                                                                                              while (k_it != it.second)

                                                                                                                              {

                                                                                                                              if (k_it->second >= start && k_it->second <= end)

                                                                                                                              break;

                                                                                                                              k_it++;

                                                                                                                              }

                                                                                                                              center = k_it->second;

                                                                                                                              (*cur)++;

                                                                                                                              temp->left = inorder_preorder(cur, umap, pre, start, center - 1);

                                                                                                                              temp->right = inorder_preorder(cur, umap, pre, center + 1, end);

                                                                                                                              return temp;

                                                                                                                              }

                                                                                                                                                          nodeptr tree(char* inorder,char*preorder,int strt,int end)
                                                                                                                                                          {
                                                                                                                                                          if(end-strt==-1)
                                                                                                                                                          return NULL;
                                                                                                                                                          static int no;
                                                                                                                                                          nodeptr node=getnode();
                                                                                                                                                          node->data=preorder[no];
                                                                                                                                                          int i=0;
                                                                                                                                                          for(i=strt;i<=end;i++)
                                                                                                                                                          {
                                                                                                                                                          if(preorder[no]==inorder[i])
                                                                                                                                                          break;
                                                                                                                                                          }
                                                                                                                                                          no++;
                                                                                                                                                          node->left=tree(inorder,preorder,strt,i-1);
                                                                                                                                                          node->right=tree(inorder,preorder,i+1,end);
                                                                                                                                                          return node;
                                                                                                                                                          }
